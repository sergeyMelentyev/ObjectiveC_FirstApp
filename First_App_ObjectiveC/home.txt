// Open Safary and show a web page
-(IBAction) URLLinker: (id) sender {
  [[UIApplication sharedApplication] openURL: [NSURL URLWithString: @"http:www.apple.com"]];
}

// Delayed action call
-(IBAction) DelayedAction: (id) sender {
  [self performSelector: @selector(AnyName) withObject: nil afterDelay: 5.0];
}

// Tag ID Sender from lots of buttons
-(IBAction) TagSenderAsArgument: (id) sender {
  currentNumber = currentNumber * 10 + (float)[sender tag];
  self.Label.text = [NSString stringWithFormat: @"2.0f", currentNumber];
}
